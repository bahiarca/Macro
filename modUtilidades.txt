Option Explicit

Function BuscarColumna(ws As Worksheet, Texto As String) As Long

    Dim filaEncabezado As Long
    Dim ultimaCol As Long
    Dim c As Long

    filaEncabezado = BuscarFilaEncabezado(ws)

    ultimaCol = ws.Cells(filaEncabezado, ws.Columns.Count).End(xlToLeft).Column

    For c = 1 To ultimaCol

        If InStr(1, UCase(Trim(ws.Cells(filaEncabezado, c).Value)), _
                 UCase(Texto)) > 0 Then

            BuscarColumna = c
            Exit Function

        End If

    Next c

    MsgBox "No se encontró la columna '" & Texto & "'"

    End

End Function

Function NormalizarDesignators(txt As String) As String

    Dim v() As String
    Dim i As Long, j As Long
    Dim aux As String

    txt = Replace(txt, " ", "")    'Eliminar espacios

    If Len(txt) = 0 Then
        NormalizarDesignators = ""
        Exit Function
    End If

    v = Split(txt, ",")

    'Ordenamiento alfabético (Bubble Sort)
    For i = LBound(v) To UBound(v) - 1
        For j = i + 1 To UBound(v)
            If UCase(v(i)) > UCase(v(j)) Then
                aux = v(i)
                v(i) = v(j)
                v(j) = aux
            End If
        Next j
    Next i

    NormalizarDesignators = Join(v, ",")

End Function
